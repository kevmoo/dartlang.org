import 'dart:async';
import 'dart:math' show Random, pow;

main() async {
 var count = 10000000;
 print('Compute π using the Monte Carlo method with '
       '$count samples.');
 var pi = await computePi(count);
 print("π ≅ $pi");
}

Future computePi(int samples, {int seed}) async {
 var points = generateRandomPoints(seed).take(samples);
 var pointsInside = points.where((p) => p.isInsideUnitCircle);
 return pointsInside.length / samples * 4;
}

Iterable generateRandomPoints([int seed]) sync* {
 var random = new Random(seed);
 while (true) {
   yield new Point(random.nextDouble(), random.nextDouble());
 }
}

class Point {
 final double x, y;
 Point(this.x, this.y);
 bool get isInsideUnitCircle => pow(x, 2) + pow(y, 2) <= 1;
}
